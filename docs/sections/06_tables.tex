\section{Índice de tablas}

\begin{table}[h]
    \centering
    \begin{tabular}{ll}
        \toprule
        Símbolo                              & Significado                                                                            \\
        \midrule
        $t \in \{0,1,2,\dots\}$              & Índice del turno.                                                                      \\
        $\cS$                                & Espacio de estados completo oculto de la batalla.                                      \\
        $s_t \in \cS$                        & Estado markoviano completo en el turno $t$.                                            \\
        $i \in \{1,2\}$                      & Índice del jugador.                                                                    \\
        $\cA_i(s)$                           & Conjunto de acciones legales para el jugador $i$ en el estado $s$.                     \\
        $a_t^i \in \cA_i(s_t)$               & Acción elegida por el jugador $i$ en el turno $t$.                                     \\
        $\cO_i$                              & Espacio de observaciones para el jugador $i$.                                          \\
        $o_t^i \in \cO_i$                    & Observación recibida por el jugador $i$ en el turno $t$.                               \\
        $\Omega_i(o\mid s)$                  & Núcleo de observación para el jugador $i$.                                             \\
        $P(s'\mid s,a^1,a^2)$                & Núcleo de transición de estado del motor del juego.                                    \\
        $h_t^i$                              & Historia de información del jugador $i$ hasta el turno $t$.                            \\
        $\pi_i(a\mid h_t^i)$                 & Política del jugador $i$.                                                              \\
        $r_i(s_t,a_t^1,a_t^2,s_{t+1})$       & Recompensa por turno para el jugador $i$.                                              \\
        $G_t^i$                              & Retorno descontado para el jugador $i$ comenzando en el turno $t$.                     \\
        $\gamma \in (0,1]$                   & Factor de descuento.                                                                   \\
        $b_t^i$                              & Estado de creencias del jugador $i$ habiéndose dados los estados ocultos dado $h_t^i$. \\
        $V^{\pi_1,\pi_2}$, $Q^{\pi_1,\pi_2}$ & Funciones de valor y valor-acción.                                                     \\
        $\Xi$                                & Conjunto de etiquetas de especies o tipos.                                             \\
        $\cM$                                & Espacio de configuraciones latentes de movimientos.                                    \\
        $\cU$                                & Espacio de configuraciones latentes de objetos/habilidades/estadísticas.               \\
        $\mathsf{HP}$                        & Variables de puntos de vida para todas las unidades activas y de reserva.              \\
        $\mathsf{Field}$                     & Estado público del campo (clima, terreno, trampas, pantallas, contadores, etc.).       \\
        $\mathsf{Priv}_i$                    & Información privada oculta del jugador $i$.                                            \\
        $\mathsf{Pub}$                       & Información de la batalla visible públicamente.                                        \\
        \bottomrule
    \end{tabular}
    \caption{Variables y espacios principales usados en la formulación completa de aprendizaje por refuerzo al estilo Pokémon}
    \label{tab:caracteristicas}
\end{table}
