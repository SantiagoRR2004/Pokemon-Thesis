\section*{\lang{Dedication}{Dedicatoria}}
\addcontentsline{toc}{section}{\lang{Dedication}{Dedicatoria}}

\newlength{\poetrywidth}
\settowidth{\poetrywidth}{remember the beast cannot be hunted alone}

\vspace*{\fill}
\begin{center}
    \begin{minipage}{\poetrywidth}
        \raggedright

        Light and dark\\
        Is that everything?\\
        An endless circle?\\
        Should the light never dim?\\
        Or should the light rebrighten the dark?

        \vspace{\baselineskip}
        \vspace{\baselineskip}

        Black holes aren’t forever\\
        A bottomless pit,\\
        \hfill A LIE\\
        Eternal light, my beloved,\\
        \hfill A LIE

        \vspace{\baselineskip}
        \vspace{\baselineskip}

        If the black dog can never be fully killed,\\
        remember the beast cannot be hunted alone
    \end{minipage}
\end{center}
\vspace*{\fill}
